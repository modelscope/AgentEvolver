 27class Tracking:
 28    """A unified tracking interface for logging experiment data to multiple backends.
 29
 30    This class provides a centralized way to log experiment metrics, parameters, and artifacts
 31    to various tracking backends including WandB, MLflow, SwanLab, TensorBoard, and console.
 32
 33    Attributes:
 34        supported_backend: List of supported tracking backends.
 35        logger: Dictionary of initialized logger instances for each backend.
 36    """
 37
 38    supported_backend = ["wandb", "mlflow", "swanlab", "vemlp_wandb", "tensorboard", "console", "clearml"]
 39
 40    def __init__(self, project_name, experiment_name, default_backend: Union[str, List[str]] = "console", config=None):
 41        if isinstance(default_backend, str):
 42            default_backend = [default_backend]
 43        for backend in default_backend:
 44            if backend == "tracking":
 45                import warnings
 46
 47                warnings.warn("`tracking` logger is deprecated. use `wandb` instead.", DeprecationWarning, stacklevel=2)
 48            else:
 49                assert backend in self.supported_backend, f"{backend} is not supported"
 50
 51        self.logger = {}
 52
 53        if "tracking" in default_backend or "wandb" in default_backend:
 54            import wandb
 55
 56            settings = None
 57            if config and config["trainer"].get("wandb_proxy", None):
 58                settings = wandb.Settings(https_proxy=config["trainer"]["wandb_proxy"])
 59            wandb.init(project=project_name, name=experiment_name, config=config, settings=settings)
 60            self.logger["wandb"] = wandb
 61
 62        if "mlflow" in default_backend:
 63            import os
 64
 65            import mlflow
 66
 67            MLFLOW_TRACKING_URI = os.environ.get("MLFLOW_TRACKING_URI", None)
 68            if MLFLOW_TRACKING_URI:
 69                mlflow.set_tracking_uri(MLFLOW_TRACKING_URI)
 70
 71            # Project_name is actually experiment_name in MLFlow
 72            # If experiment does not exist, will create a new experiment
 73            experiment = mlflow.set_experiment(project_name)
 74            mlflow.start_run(experiment_id=experiment.experiment_id, run_name=experiment_name)
 75            mlflow.log_params(_compute_mlflow_params_from_objects(config))
 76            self.logger["mlflow"] = _MlflowLoggingAdapter()
 77
 78        if "swanlab" in default_backend:
 79            import os
 80
 81            import swanlab
 82
 83            SWANLAB_API_KEY = os.environ.get("SWANLAB_API_KEY", None)
 84            SWANLAB_LOG_DIR = os.environ.get("SWANLAB_LOG_DIR", "swanlog")
 85            SWANLAB_MODE = os.environ.get("SWANLAB_MODE", "cloud")
 86            if SWANLAB_API_KEY:
 87                swanlab.login(SWANLAB_API_KEY)  # NOTE: previous login information will be overwritten
 88
 89            if config is None:
 90                config = {}  # make sure config is not None, otherwise **config will raise error
 91            swanlab.init(
 92                project=project_name,
 93                experiment_name=experiment_name,
 94                config={"FRAMEWORK": "verl", **config},
 95                logdir=SWANLAB_LOG_DIR,
 96                mode=SWANLAB_MODE,
 97                settings=swanlab.Settings(backup=False, log_proxy_type='none')
 98            )
 99            self.logger["swanlab"] = swanlab
100
101        if "vemlp_wandb" in default_backend:
102            import os
103
104            import volcengine_ml_platform
105            from volcengine_ml_platform import wandb as vemlp_wandb
106
107            volcengine_ml_platform.init(
108                ak=os.environ["VOLC_ACCESS_KEY_ID"],
109                sk=os.environ["VOLC_SECRET_ACCESS_KEY"],
110                region=os.environ["MLP_TRACKING_REGION"],
111            )
112
113            vemlp_wandb.init(
114                project=project_name,
115                name=experiment_name,
116                config=config,
117                sync_tensorboard=True,
118            )
119            self.logger["vemlp_wandb"] = vemlp_wandb
120
121        if "tensorboard" in default_backend:
122            self.logger["tensorboard"] = _TensorboardAdapter()
123
124        if "console" in default_backend:
125            from verl.utils.logger import LocalLogger
126
127            self.console_logger = LocalLogger(print_to_console=True)
128            self.logger["console"] = self.console_logger
129
130        if "clearml" in default_backend:
131            self.logger["clearml"] = ClearMLLogger(project_name, experiment_name, config)
132
133    def log(self, data, step, backend=None):
134        for default_backend, logger_instance in self.logger.items():
135            if backend is None or default_backend in backend:
136                logger_instance.log(data=data, step=step)
137
138    def __del__(self):
139        if "wandb" in self.logger:
140            self.logger["wandb"].finish(exit_code=0)
141        if "swanlab" in self.logger:
142            self.logger["swanlab"].finish()
143        if "vemlp_wandb" in self.logger:
144            self.logger["vemlp_wandb"].finish(exit_code=0)
145        if "tensorboard" in self.logger:
146            self.logger["tensorboard"].finish()
147
148        if "clearnml" in self.logger:
149            self.logger["clearnml"].finish()

152class ClearMLLogger:
153    def __init__(self, project_name: str, experiment_name: str, config):
154        self.project_name = project_name
155        self.experiment_name = experiment_name
156
157        import clearml
158
159        self._task: clearml.Task = clearml.Task.init(
160            task_name=experiment_name,
161            project_name=project_name,
162            continue_last_task=True,
163            output_uri=False,
164        )
165
166        self._task.connect_configuration(config, name="Hyperparameters")
167
168    def _get_logger(self):
169        return self._task.get_logger()
170
171    def log(self, data, step):
172        import numpy as np
173        import pandas as pd
174
175        # logs = self._rewrite_logs(data)
176        logger = self._get_logger()
177        for k, v in data.items():
178            title, series = k.split("/", 1)
179
180            if isinstance(v, (int, float, np.floating, np.integer)):
181                logger.report_scalar(
182                    title=title,
183                    series=series,
184                    value=v,
185                    iteration=step,
186                )
187            elif isinstance(v, pd.DataFrame):
188                logger.report_table(
189                    title=title,
190                    series=series,
191                    table_plot=v,
192                    iteration=step,
193                )
194            else:
195                logger.warning(f'Trainer is attempting to log a value of "{v}" of type {type(v)} for key "{k}". This invocation of ClearML logger\'s function is incorrect so this attribute was dropped. ')
196
197    def finish(self):
198        self._task.mark_completed()

202    def __init__(self):
203        import os
204

228def _compute_mlflow_params_from_objects(params) -> Dict[str, Any]:
229    if params is None:
230        return {}
231
232    return _flatten_dict(_transform_params_to_json_serializable(params, convert_list_to_dict=True), sep="/")
233
234
235def _transform_params_to_json_serializable(x, convert_list_to_dict: bool):
236    _transform = partial(_transform_params_to_json_serializable, convert_list_to_dict=convert_list_to_dict)
237
238    if dataclasses.is_dataclass(x):
239        return _transform(dataclasses.asdict(x))
240    if isinstance(x, dict):
241        return {k: _transform(v) for k, v in x.items()}
242    if isinstance(x, list):
243        if convert_list_to_dict:
244            return {"list_len": len(x)} | {f"{i}": _transform(v) for i, v in enumerate(x)}
245        else:
246            return [_transform(v) for v in x]
247    if isinstance(x, Path):
248        return str(x)
249    if isinstance(x, Enum):
250        return x.value
251
252    return x

255def _flatten_dict(raw: Dict[str, Any], *, sep: str) -> Dict[str, Any]:
256    import pandas as pd
257
258    ans = pd.json_normalize(raw, sep=sep).to_dict(orient="records")[0]
259    assert isinstance(ans, dict)
260    return ans

264class ValidationGenerationsLogger:
265    def log(self, loggers, samples, step):
266        if "wandb" in loggers:
267            self.log_generations_to_wandb(samples, step)
268        if "swanlab" in loggers:
269            self.log_generations_to_swanlab(samples, step)
270        if "mlflow" in loggers:
271            self.log_generations_to_mlflow(samples, step)
272
273        if "clearml" in loggers:
274            self.log_generations_to_clearml(samples, step)
275        if "tensorboard" in loggers:
276            self.log_generations_to_tensorboard(samples, step)

278    def log_generations_to_wandb(self, samples, step):
279        """Log samples to wandb as a table"""
280        import wandb
281
282        # Create column names for all samples
283        columns = ["step"] + sum([[f"input_{i + 1}", f"output_{i + 1}", f"score_{i + 1}"] for i in range(len(samples))], [])
284
285        if not hasattr(self, "validation_table"):
286            # Initialize the table on first call
287            self.validation_table = wandb.Table(columns=columns)
288
289        # Create a new table with same columns and existing data
290        # Workaround for https://github.com/wandb/wandb/issues/2981#issuecomment-1997445737
291        new_table = wandb.Table(columns=columns, data=self.validation_table.data)
292
293        # Add new row with all data
294        row_data = []
295        row_data.append(step)
296        for sample in samples:
297            row_data.extend(sample)
298
299        new_table.add_data(*row_data)
300
301        # Update reference and log
302        wandb.log({"val/generations": new_table}, step=step)
303        self.validation_table = new_table
304
305    def log_generations_to_swanlab(self, samples, step):
306        """Log samples to swanlab as text"""
307        import swanlab
308
309        swanlab_text_list = []
310        for i, sample in enumerate(samples):
311            row_text = f"""
312            input: {sample[0]}
313            
314            ---
315            
316            output: {sample[1]}
317            
318            ---
319            
320            score: {sample[2]}
321            """
322            swanlab_text_list.append(swanlab.Text(row_text, caption=f"sample {i + 1}"))
323
324        # Log to swanlab
325        swanlab.log({"val/generations": swanlab_text_list}, step=step)
326
327    def log_generations_to_mlflow(self, samples, step):
328        """Log validation generation to mlflow as artifacts"""
329        # https://mlflow.org/docs/latest/api_reference/python_api/mlflow.html?highlight=log_artifact#mlflow.log_artifact
330
331        import json
332        import tempfile
333
334        import mlflow
335
336        try:
337            with tempfile.TemporaryDirectory() as tmp_dir:
338                validation_gen_step_file = Path(tmp_dir, f"val_step{step}.json")
339                row_data = []
340                for sample in samples:
341                    data = {"input": sample[0], "output": sample[1], "score": sample[2]}
342                    row_data.append(data)
343                with open(validation_gen_step_file, "w") as file:
344                    json.dump(row_data, file)
345                mlflow.log_artifact(validation_gen_step_file)
346        except Exception as e:
347            print(f"WARNING: save validation generation file to mlflow failed with error {e}")
348
349    def log_generations_to_clearml(self, samples, step):
350        """Log validation generation to clearml as table"""
351
352        import clearml
353        import pandas as pd
354
355        task: clearml.Task | None = clearml.Task.current_task()
356        if task is None:
357            return
358
359        table = [
360            {
361                "step": step,
362                "input": sample[0],
363                "output": sample[1],
364                "score": sample[2],
365            }
366            for sample in samples
367        ]
368
369        logger = task.get_logger()
370        logger.report_table(
371            series="Validation generations",
372            title="Validation",
373            table_plot=pd.DataFrame.from_records(table),
374            iteration=step,
375        )
376
377    def log_generations_to_tensorboard(self, samples, step):
378        """Log samples to tensorboard as text"""
379        # Initialize tensorboard writer if not exists
380        if not hasattr(self, "writer"):
381            from torch.utils.tensorboard import SummaryWriter
382
383            tensorboard_dir = os.environ.get("TENSORBOARD_DIR", "tensorboard_log")
384            os.makedirs(tensorboard_dir, exist_ok=True)
385            self.writer = SummaryWriter(log_dir=tensorboard_dir)
386
387        # Format the samples data into readable text
388        text_content = f"**Generation Results - Step {step}**\n\n"
389
390        for i, sample in enumerate(samples):
391            text_content += f"### Sample {i + 1}\n"
392
393            # Assuming sample contains [input, output, score]
394            if len(sample) >= 3:
395                input_text, output_text, score = sample[0], sample[1], sample[2]
396
397                text_content += f"**Input:** {input_text}\n\n"
398                text_content += f"**Output:** {output_text}\n\n"
399                text_content += f"**Score:** {score}\n\n"
400            else:
401                # Handle cases where sample format might be different
402                text_content += f"**Data:** {sample}\n\n"
403
404            text_content += "---\n\n"
405
406        # Log to tensorboard as text
407        self.writer.add_text("val/generations", text_content, step)
408        # Flush to ensure data is written
409        self.writer.flush()

280        """Log samples to wandb as a table"""
281        import wandb
282
283        # Create column names for all samples
284        columns = ["step"] + sum([[f"input_{i + 1}", f"output_{i + 1}", f"score_{i + 1}"] for i in range(len(samples))], [])
285
286        if not hasattr(self, "validation_table"):
287            # Initialize the table on first call

291        # Workaround for https://github.com/wandb/wandb/issues/2981#issuecomment-1997445737

293
294        # Add new row with all data
295        row_data = []

297        for sample in samples:
298            row_data.extend(sample)
299
300        new_table.add_data(*row_data)
301
302        # Update reference and log
303        wandb.log({"val/generations": new_table}, step=step)
304        self.validation_table = new_table
