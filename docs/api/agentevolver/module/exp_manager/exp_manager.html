 56    def summarize_in_batch(self, trajectories: List[Trajectory]) -> None:
 57        trajectories_sorted = sorted(trajectories, key=lambda traj: traj.task_id)
 58        grouped_trajectories = [list(group) for key, group in groupby(trajectories_sorted, key=lambda traj: traj.task_id)]
 59        batch_size = self.exp_manager_config.summary_batch_size
 60        all_batches = []
 61        for group in grouped_trajectories:
 62            for i in range(0, len(group), batch_size):
 63                all_batches.append(group[i:i + batch_size])
 64        
 65        futures = []
 66        for batch in all_batches:
 67            future = self.thread_pool.submit(
 68                self.em_client.call_summarizer,
 69                trajectories=batch,
 70                workspace_id=self.reme_config.workspace_id
 71            )
 72            futures.append(future)
 73        
 74        results = []
 75        for future in as_completed(futures):
 76            try:
 77                result = future.result()
 78                results.append(result)
 79            except Exception as e:
 80                print(f"Error in summary task: {e}")
 81        
 82        return

167    def allocate_train_mode(self, tasks: List[Task]) -> List[TaskExpConfig]:
168        """
169        Allocates training modes for the given tasks based on the configured training sample experience mode.
170
171        Args:
172            tasks (List[Task]): A list of Task objects for which to allocate training modes.
173
174        Returns:
175            List[TaskExpConfig]: A list of TaskExpConfig objects with allocated training modes.
176        """
177        mode_to_ratio = {
178            "allkeep": 1.0,
179            "alldiscard": 0.0,
180            "hybrid": self.train_sample_keepratio
181        }
182        keep_ratio = mode_to_ratio.get(
183            self.train_sample_mode, self.train_sample_keepratio
184        )
185        keep_count = int(len(tasks) * keep_ratio)
186        exp_modes = ['keep'] * keep_count + ['discard'] * (len(tasks) - keep_count)
187        random.shuffle(exp_modes)
188        return [TaskExpConfig(add_exp=[], train_mode=exp_mode) for exp_mode in exp_modes]

190    def allocate_add_exp(self, exp_configs: List[TaskExpConfig], mode: Literal["sample", "validate"]) -> List[TaskExpConfig]:
191        """
192        Allocates experience addition settings for the given tasks based on the mode and configured experience modes.
193
194        Args:
195            exp_configs (List[TaskExpConfig]): A list of TaskExpConfig objects to be updated.
196            mode (Literal["sample", "validate"]): The mode of operation, either "sample" or "validate".
197
198        Returns:
199            List[TaskExpConfig]: An updated list of TaskExpConfig objects with allocated experience addition settings.
200        """
201        is_validate = mode == "validate"
202        rollout_n = self.rollout_config.val_kwargs.n if is_validate else self.rollout_config.n
203        exp_mode = self.val_rollout_mode if is_validate else self.train_rollout_mode
204        for task_exp_config in exp_configs:
205            add_exp_choices = {
206                "woexp": [False] * rollout_n,
207                "mixed": sorted([i < round(rollout_n*self.rollout_ratio) for i in range(rollout_n)], key=lambda _: random.random()),
208                "all": [True] * rollout_n
209            }[exp_mode]
210            task_exp_config.add_exp = add_exp_choices
211        
212        return exp_configs

228    def manage_rollout_context(self, init_messages: List[dict], traj_exp_config: TrajExpConfig) -> Tuple[List[dict], TrajExpConfig]:
229        """
230        Manages the context for the rollout phase, potentially adding historical experience.
231
232        Args:
233            init_messages (List[dict]): Initial messages for the rollout.
234            traj_exp_config (TrajExpConfig): Configuration for the trajectory experience.
235
236        Returns:
237            Tuple[List[dict], TrajExpConfig]: Updated messages and modified trajectory experience config.
238        """
239        # check experience conditions
240        if not self._should_process_experience(traj_exp_config):
241            return init_messages, traj_exp_config
242        
243        # initialize em client
244        self._ensure_em_client()
245        
246        # construct trajectory
247        trajectory = Trajectory(
248            data_id=traj_exp_config.data_id,
249            rollout_id=traj_exp_config.rollout_id,
250            steps=init_messages,
251            query=traj_exp_config.query
252        )
253
254        # retrieve experience
255        reme_config = self.config.exp_manager.reme
256        history_experience = self.em_client.call_context_generator(
257            trajectory=trajectory,
258            retrieve_top_k=reme_config.retrieve_top_k,
259            workspace_id=reme_config.workspace_id
260        )
261
262        # check empty condition
263        if not history_experience:
264            logger.info("Experience is empty!")
265            return init_messages, traj_exp_config
266
267        # apply experience to trajectory
268        logger.info(f"Retrieved history experience: {history_experience}")
269        formatted_experience = self.experience_template.format(history_experience)
270        new_content = formatted_experience + trajectory.steps[-1]["content"]
271        trajectory.steps[-1]["content"] = new_content
272        traj_exp_config.experience_list = traj_exp_config.experience_list + [formatted_experience]
273
274        return trajectory.steps, traj_exp_config

276    def _should_process_experience(self, traj_exp_config: TrajExpConfig) -> bool:
277        """
278        Checks if experience processing should be performed.
279
280        Args:
281            traj_exp_config (TrajExpConfig): Configuration for the trajectory experience.
282
283        Returns:
284            bool: True if experience should be processed, False otherwise.
285        """
286        return (traj_exp_config.add_exp and
287                self.config.exp_manager.reme.enable_context_generator)

294            self.em_client = EMClient(

300    def manage_training_context(self, message: str, metadata_config: Dict) -> Tuple[str, str]:
301        """
302        Extracts and removes experience information from the given message.
303
304        Args:
305            message (str): Input message potentially containing experience information.
306            metadata_config (Dict): Configuration for the trajectory experience.
307
308        Returns:
309            Tuple[str, str]: Extracted experience and the message with experience information removed.
310        """
311        experience = ""
312        cleaned_message = message
313
314        if metadata_config.get("task_train_mode", "discard") == "discard": 
315            pattern = re.escape(self.experience_template).replace(r'\{\}', '(.*?)')
316            match = re.search(pattern, message, re.DOTALL)
317            if match:
318                experience = match.group(1)
319                cleaned_message = re.sub(pattern, '', message, flags=re.DOTALL)
320
321        
322        return experience, cleaned_message

168    def allocate_train_mode(self, tasks: List[Task]) -> List[TaskExpConfig]:
169        """
170        Allocates training modes for the given tasks based on the configured training sample experience mode.
171
172        Args:
173            tasks (List[Task]): A list of Task objects for which to allocate training modes.
174
175        Returns:
176            List[TaskExpConfig]: A list of TaskExpConfig objects with allocated training modes.
177        """
178        mode_to_ratio = {
179            "allkeep": 1.0,
180            "alldiscard": 0.0,
181            "hybrid": self.train_sample_keepratio
182        }
183        keep_ratio = mode_to_ratio.get(
184            self.train_sample_mode, self.train_sample_keepratio
185        )
186        keep_count = int(len(tasks) * keep_ratio)
187        exp_modes = ['keep'] * keep_count + ['discard'] * (len(tasks) - keep_count)
188        random.shuffle(exp_modes)
189        return [TaskExpConfig(add_exp=[], train_mode=exp_mode) for exp_mode in exp_modes]

192        """
193        Allocates experience addition settings for the given tasks based on the mode and configured experience modes.
194
