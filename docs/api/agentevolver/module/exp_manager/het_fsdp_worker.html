  75def create_device_mesh(world_size, fsdp_size):
  76    """
  77    Initializes a device mesh for distributed training.
  78
  79    Args:
  80        world_size (int): The total number of processes in the distributed training setup.
  81        fsdp_size (int): The number of processes to be used for FSDP sharding.
  82
  83    Returns:
  84        torch.distributed.device_mesh.DeviceMesh: The initialized device mesh.
  85    """
  86    if fsdp_size < 0 or fsdp_size >= world_size:
  87        device_mesh = init_device_mesh(device_name, mesh_shape=(world_size,), mesh_dim_names=["fsdp"])  # ⭐ Initialize device mesh for FSDP
  88    else:
  89        device_mesh = init_device_mesh(device_name, mesh_shape=(world_size // fsdp_size, fsdp_size), mesh_dim_names=["ddp", "fsdp"])  # ⭐ Initialize device mesh for DDP and FSDP
  90    return device_mesh

  93def get_sharding_strategy(device_mesh):
  94    """
  95    Determines the sharding strategy based on the dimensionality of the device mesh.
  96
  97    Args:
  98        device_mesh (torch.Tensor): The device mesh used for sharding.
  99
 100    Returns:
 101        ShardingStrategy: The sharding strategy to be used.
 102    """
 103    from torch.distributed.fsdp import ShardingStrategy
 104
 105    if device_mesh.ndim == 1:
 106        sharding_strategy = ShardingStrategy.FULL_SHARD  # ⭐ Set sharding strategy for 1D device mesh
 107    elif device_mesh.ndim == 2:
 108        sharding_strategy = ShardingStrategy.HYBRID_SHARD  # ⭐ Set sharding strategy for 2D device mesh
 109    else:
 110        raise NotImplementedError(f"Get device mesh ndim={device_mesh.ndim}, but only support 1 or 2")
 111    return sharding_strategy

 176                self.config.actor.ppo_micro_batch_size //= self.device_mesh.size() // self.ulysses_sequence_parallel_size

 186            self.config.rollout.log_prob_micro_batch_size_per_gpu = self.config.rollout.log_prob_micro_batch_size

 192    def _build_model_optimizer(
 193        self,
 194        model_path,
 195        fsdp_config,
 196        optim_config,
 197        override_model_config,
 198        use_remove_padding=False,
 199        use_fused_kernels=False,
 200        enable_gradient_checkpointing=False,
 201        trust_remote_code=False,
 202        use_liger=False,
 203        role="actor",
 204        enable_activation_offload=False,
 205    ):
 206        """
 207        Builds and configures the model and its optimizer for a distributed training setup using FSDP.
 208
 209        Args:
 210            model_path (str): The path to the model.
 211            fsdp_config (dict): Configuration for FSDP.
 212            optim_config (dict): Configuration for the optimizer.
 213            override_model_config (dict): Additional configuration to override the model's default settings.
 214            use_remove_padding (bool, optional): Whether to remove padding. Defaults to False.
 215            use_fused_kernels (bool, optional): Whether to use fused kernels. Defaults to False.
 216            enable_gradient_checkpointing (bool, optional): Whether to enable gradient checkpointing. Defaults to False.
 217            trust_remote_code (bool, optional): Whether to trust remote code. Defaults to False.
 218            use_liger (bool, optional): Whether to apply Liger kernel to the model. Defaults to False.
 219            role (str, optional): The role of the model (e.g., "actor", "ref"). Defaults to "actor".
 220            enable_activation_offload (bool, optional): Whether to enable activation offloading. Defaults to False.
 221
 222        Returns:
 223            None
 224        """
 225        from torch import optim
 226        from torch.distributed.fsdp import CPUOffload, MixedPrecision
 227        from transformers import AutoConfig, AutoModelForCausalLM, AutoModelForVision2Seq
 228
 229        from verl.utils.model import get_generation_config, print_model_size, update_model_config
 230        from verl.utils.torch_dtypes import PrecisionType
 231
 232        assert role in ["actor", "ref"]  # ⭐ Ensures the role is either "actor" or "ref"
 233
 234        log_gpu_memory_usage(f"Before init {role} from HF AutoModel", logger=logger)
 235        local_path = model_path
 236
 237        # note that we have to create model in fp32. Otherwise, the optimizer is in bf16, which is incorrect
 238        # TODO(zhangchi.usc1992): 1. support create from random initialized model. 2. Support init with FSDP directly
 239        self.tokenizer = hf_tokenizer(local_path, trust_remote_code=trust_remote_code)
 240        self.processor = hf_processor(local_path, trust_remote_code=trust_remote_code)
 241
 242        torch_dtype = fsdp_config.get("model_dtype", None)
 243        if torch_dtype is None:
 244            torch_dtype = torch.float32 if self._is_actor else torch.bfloat16
 245        else:
 246            torch_dtype = PrecisionType.to_dtype(torch_dtype)
 247
 248        # override model kwargs
 249        actor_model_config = AutoConfig.from_pretrained(local_path, trust_remote_code=trust_remote_code, attn_implementation="flash_attention_2")
 250
 251        # patch for kimi-vl
 252        if getattr(actor_model_config, "model_type", None) == "kimi_vl":
 253            actor_model_config.text_config.topk_method = "greedy"
 254
 255        self.generation_config = get_generation_config(local_path, trust_remote_code=trust_remote_code)
 256
 257        override_config_kwargs = {
 258            "bos_token_id": self.tokenizer.bos_token_id,
 259            "eos_token_id": self.tokenizer.eos_token_id,
 260            "pad_token_id": self.tokenizer.pad_token_id,
 261        }
 262        override_config_kwargs.update(override_model_config)
 263        update_model_config(actor_model_config, override_config_kwargs=override_config_kwargs)
 264        if self.rank == 0:
 265            print(f"Model config after override: {actor_model_config}")
 266
 267        # NOTE(fix me): tie_word_embedding causes meta_tensor init to hang
 268        init_context = get_init_weight_context_manager(use_meta_tensor=not actor_model_config.tie_word_embeddings, mesh=self.device_mesh)
 269
 270        with init_context(), warnings.catch_warnings():
 271            warnings.simplefilter("ignore")
 272            if type(actor_model_config) in AutoModelForVision2Seq._model_mapping.keys():
 273                actor_module_class = AutoModelForVision2Seq
 274            else:
 275                actor_module_class = AutoModelForCausalLM
 276
 277            actor_module = actor_module_class.from_pretrained(
 278                pretrained_model_name_or_path=local_path,
 279                torch_dtype=torch_dtype,
 280                config=actor_model_config,
 281                trust_remote_code=trust_remote_code,
 282            )  # ⭐ Loads the model from the specified path with the given configuration
 283
 284            # Apply Liger kernel to the model if use_liger is set to True
 285            if use_liger:
 286                from liger_kernel.transformers.monkey_patch import _apply_liger_kernel_to_instance
 287
 288                _apply_liger_kernel_to_instance(model=actor_module)
 289
 290            fused_kernel_options = self.config.model.get("fused_kernel_options", None)
 291            fused_kernels_backend = fused_kernel_options.get("impl_backend", None) if fused_kernel_options is not None else None
 292
 293            apply_monkey_patch(
 294                model=actor_module,
 295                use_remove_padding=use_remove_padding,
 296                ulysses_sp_size=self.ulysses_sequence_parallel_size,
 297                use_fused_kernels=use_fused_kernels,
 298                fused_kernels_backend=fused_kernels_backend,
 299            )
 300
 301            # some parameters may not in torch_dtype. TODO(zhangchi.usc1992) remove this after we switch to fsdp2
 302            actor_module.to(torch_dtype)
 303
 304            if enable_gradient_checkpointing:
 305                actor_module.gradient_checkpointing_enable(gradient_checkpointing_kwargs={"use_reentrant": False})  # ⭐ Enables gradient checkpointing for the model
 306
 307            if self._is_lora:
 308                print("Applying LoRA to actor module")
 309                actor_module.enable_input_require_grads()
 310                # Convert config to regular Python types before creating PEFT model
 311                lora_config = {"task_type": TaskType.CAUSAL_LM, "r": self.config.model.lora_rank, "lora_alpha": self.config.model.lora_alpha, "target_modules": convert_to_regular_types(self.config.model.target_modules), "bias": "none"}
 312                actor_module = get_peft_model(actor_module, LoraConfig(**lora_config))
 313        torch.distributed.barrier()
 314
 315        if self.rank == 0:
 316            print_model_size(actor_module)
 317
 318        log_gpu_memory_usage(f"After init {role} from HF AutoModel", logger=logger)
 319
 320        # We wrap FSDP for rollout as well
 321        mixed_precision_config = fsdp_config.get("mixed_precision", None)
 322        if mixed_precision_config is not None:
 323            param_dtype = PrecisionType.to_dtype(mixed_precision_config.get("param_dtype", "bf16"))
 324            reduce_dtype = PrecisionType.to_dtype(mixed_precision_config.get("reduce_dtype", "fp32"))
 325            buffer_dtype = PrecisionType.to_dtype(mixed_precision_config.get("buffer_dtype", "fp32"))
 326        else:
 327            param_dtype = torch.bfloat16
 328            reduce_dtype = torch.float32
 329            buffer_dtype = torch.float32
 330
 331        mixed_precision = MixedPrecision(param_dtype=param_dtype, reduce_dtype=reduce_dtype, buffer_dtype=buffer_dtype)
 332
 333        auto_wrap_policy = get_fsdp_wrap_policy(module=actor_module, config=fsdp_config.get("wrap_policy", None), is_lora=self.config.model.get("lora_rank", 0) > 0)
 334
 335        if self._is_rollout and self.config.rollout.name == "hf":
 336            # TODO(zhangchi.usc1992, shengguangming) fix me. Current, auto_wrap_policy causes HFRollout to hang in Gemma
 337            auto_wrap_policy = None
 338
 339        if self.rank == 0:
 340            print(f"wrap_policy: {auto_wrap_policy}")
 341
 342        fsdp_mesh = self.device_mesh
 343        sharding_strategy = get_sharding_strategy(fsdp_mesh)
 344
 345        # We force reference policy to use CPUOffload to save memory.
 346        # We force turn off CPUOffload for actor because it causes incorrect results when using grad accumulation
 347        cpu_offload = None if role == "actor" else CPUOffload(offload_params=True)
 348        fsdp_strategy = self.config.actor.strategy
 349        if fsdp_strategy == "fsdp":
 350            actor_module_fsdp = FSDP(
 351                actor_module,
 352                cpu_offload=cpu_offload,
 353                param_init_fn=init_fn,
 354                use_orig_params=False,
 355                auto_wrap_policy=auto_wrap_policy,
 356                device_id=get_device_id(),
 357                sharding_strategy=sharding_strategy,  # zero3
 358                mixed_precision=mixed_precision,
 359                sync_module_states=True,
 360                device_mesh=self.device_mesh,
 361                forward_prefetch=self.config.actor.fsdp_config.forward_prefetch,
 362            )  # ⭐ Initialize the FSDP wrapper for the actor module
 363        elif fsdp_strategy == "fsdp2":
 364            assert CPUOffloadPolicy is not None, "PyTorch version >= 2.4 is required for using fully_shard API (FSDP2)"
 365            mp_policy = MixedPrecisionPolicy(param_dtype=param_dtype, reduce_dtype=reduce_dtype, cast_forward_inputs=True)
 366            if role == "actor" and fsdp_config.offload_policy:
 367                cpu_offload = CPUOffloadPolicy(pin_memory=True)
 368                self._is_offload_param = False
 369                self._is_offload_optimizer = False
 370            else:
 371                cpu_offload = None if role == "actor" else CPUOffloadPolicy(pin_memory=True)
 372
 373            fsdp_kwargs = {
 374                "mesh": fsdp_mesh,
 375                "mp_policy": mp_policy,
 376                "offload_policy": cpu_offload,
 377                "reshard_after_forward": fsdp_config.reshard_after_forward,
 378            }
 379            full_state = actor_module.state_dict()
 380            apply_fsdp2(actor_module, fsdp_kwargs, fsdp_config)
 381            fsdp2_load_full_state_dict(actor_module, full_state, fsdp_mesh, cpu_offload)
 382            actor_module_fsdp = actor_module
 383        else:
 384            raise NotImplementedError(f"not implement {fsdp_strategy}")
 385
 386        if enable_activation_offload:
 387            enable_activation_offloading(actor_module_fsdp, fsdp_strategy, enable_gradient_checkpointing)
 388
 389        log_gpu_memory_usage(f"After {role} FSDP init", logger=logger)
 390
 391        # TODO: add more optimizer args into config
 392        if role == "actor" and optim_config is not None:
 393            from verl.utils.torch_functional import get_constant_schedule_with_warmup, get_cosine_schedule_with_warmup
 394
 395            actor_optimizer = optim.AdamW(
 396                actor_module_fsdp.parameters(),
 397                lr=optim_config.lr,
 398                betas=optim_config.get("betas", (0.9, 0.999)),
 399                weight_decay=optim_config.get("weight_decay", 1e-2),
 400            )  # ⭐ Initialize the AdamW optimizer for the actor module
 401
 402            total_steps = optim_config.get("total_training_steps", 0)
 403            num_warmup_steps = int(optim_config.get("lr_warmup_steps", -1))
 404            warmup_style = optim_config.get("warmup_style", "constant")
 405            min_lr_ratio = optim_config.get("min_lr_ratio", 0.0)
 406            num_cycles = optim_config.get("num_cycles", 0.5)
 407            if num_warmup_steps < 0:
 408                num_warmup_steps_ratio = optim_config.get("lr_warmup_steps_ratio", 0.0)
 409                num_warmup_steps = int(num_warmup_steps_ratio * total_steps)
 410
 411            if self.rank == 0:
 412                print(f"Total steps: {total_steps}, num_warmup_steps: {num_warmup_steps}")
 413
 414            if warmup_style == "constant":
 415                actor_lr_scheduler = get_constant_schedule_with_warmup(optimizer=actor_optimizer, num_warmup_steps=num_warmup_steps)
 416            elif warmup_style == "cosine":
 417                actor_lr_scheduler = get_cosine_schedule_with_warmup(optimizer=actor_optimizer, num_warmup_steps=num_warmup_steps, num_training_steps=total_steps, min_lr_ratio=min_lr_ratio, num_cycles=num_cycles)  # ⭐ Initialize cosine learning rate scheduler
 418            else:
 419                raise NotImplementedError(f"Warmup style {warmup_style} is not supported")
 420
 421            log_gpu_memory_usage(f"After {role} optimizer init", logger=logger)
 422        else:
 423            actor_optimizer = None
 424            actor_lr_scheduler = None
 425
 426        return actor_module_fsdp, actor_optimizer, actor_lr_scheduler, actor_model_config

 428    def _build_rollout(self, trust_remote_code=False):
 429        """
 430        Builds the rollout and its associated sharding manager based on the configuration.
 431
 432        Args:
 433            trust_remote_code (bool): Whether to trust remote code or not.
 434
 435        Returns:
 436            tuple: A tuple containing the rollout and its sharding manager.
 437        """
 438        from torch.distributed.device_mesh import init_device_mesh
 439
 440        # TODO(sgm): support FSDP hybrid shard for larger model
 441        infer_tp = self.config.rollout.tensor_model_parallel_size
 442        dp = self.world_size // infer_tp
 443        assert self.world_size % infer_tp == 0, f"rollout world_size: {self.world_size} is not divisible by infer_tp: {infer_tp}"
 444        rollout_device_mesh = init_device_mesh(device_name, mesh_shape=(dp, infer_tp), mesh_dim_names=["dp", "infer_tp"])  # ⭐ Initialize the device mesh for the rollout
 445        rollout_name = self.config.rollout.name
 446        if rollout_name == "hf":
 447            from verl.workers.rollout import HFRollout
 448            from verl.workers.sharding_manager.base import BaseShardingManager
 449
 450            rollout = HFRollout(module=self.actor_module_fsdp, config=self.config.rollout)
 451            rollout_sharding_manager = BaseShardingManager()
 452            # TODO: a sharding manager that do nothing?
 453
 454        elif rollout_name == "vllm":
 455            from verl.workers.rollout.vllm_rollout import vllm_mode, vLLMRollout

 645    @register(dispatch_mode=Dispatch.DP_COMPUTE_PROTO)
 646    def update_actor(self, data: DataProto):
 647        """
 648        Updates the actor model in a distributed training setup using Fully Sharded Data Parallel (FSDP).
 649
 650        Args:
 651            data (DataProto): The input data for the actor model.
 652
 653        Returns:
 654            DataProto: The output data containing the updated metrics.
 655        """
 656        # Support all hardwares
 657        data = data.to("cpu")  # data will to device with each micro batch on actor.update_policy
 658
 659        assert self._is_actor
 660        if self._is_offload_param:
 661            load_fsdp_model_to_gpu(self.actor_module_fsdp)  # ⭐ Load the FSDP model to GPU
 662        if self._is_offload_optimizer:
 663            load_fsdp_optimizer(optimizer=self.actor_optimizer, device_id=get_device_id())  # ⭐ Load the FSDP optimizer to GPU
 664
 665        with self.ulysses_sharding_manager:
 666            data = self.ulysses_sharding_manager.preprocess_data(data=data)
 667            # perform training
 668            with Timer(name="update_policy", logger=None) as timer:
 669                metrics = self.actor.update_policy(data=data)  # ⭐ Update the actor policy with the provided data
 670            delta_time = timer.last
 671            global_num_tokens = data.meta_info["global_token_num"]
 672            estimated_flops, promised_flops = self.flops_counter.estimate_flops(global_num_tokens, delta_time)
 673            metrics["perf/mfu/actor"] = estimated_flops * self.config.actor.ppo_epochs / promised_flops / self.world_size
 674            metrics["perf/max_memory_allocated_gb"] = get_torch_device().max_memory_allocated() / (1024**3)
 675            metrics["perf/max_memory_reserved_gb"] = get_torch_device().max_memory_reserved() / (1024**3)
 676            metrics["perf/cpu_memory_used_gb"] = psutil.virtual_memory().used / (1024**3)
 677
 678            lr = self.actor_lr_scheduler.get_last_lr()[0]
 679            metrics["actor/lr"] = lr
 680            self.actor_lr_scheduler.step()
 681
 682            # TODO: here, we should return all metrics
 683            output = DataProto(meta_info={"metrics": metrics})
 684
 685            output = self.ulysses_sharding_manager.postprocess_data(data=output)
 686            output = output.to("cpu")
 687
 688        if self._is_offload_param:
 689            offload_fsdp_model_to_cpu(self.actor_module_fsdp)  # ⭐ Offload the FSDP model to CPU
 690            log_gpu_memory_usage("After offload actor model during update_actor", logger=logger)
 691        if self._is_offload_optimizer:
 692            offload_fsdp_optimizer(optimizer=self.actor_optimizer)  # ⭐ Offload the FSDP optimizer to CPU
 693            log_gpu_memory_usage("After offload actor optimizer during update_actor", logger=logger)
 694
 695        return output

 697    @register(dispatch_mode=Dispatch.DP_COMPUTE_PROTO)

 699        """
 700        Generates sequences based on the provided prompts. This function handles moving the prompts to the correct device,
 701        updating meta information, and using a rollout sharding manager to preprocess, generate, and postprocess the data.
 702
 703        Args:
 704            prompts (DataProto): The input prompts for sequence generation.
 705
 706        Returns:
 707            DataProto: The generated sequences with updated meta information.
 708        """
 709        # Support all hardwares
 710        prompts = prompts.to(get_device_id())  # ⭐ Move prompts to the correct device
 711
 712        assert self._is_rollout
 713
 714        meta_info = {
 715            "eos_token_id": self.generation_config.eos_token_id if self.generation_config is not None else self.tokenizer.eos_token_id,
 716            "pad_token_id": self.generation_config.pad_token_id if self.generation_config is not None else self.tokenizer.pad_token_id,
 717        }
 718        prompts.meta_info.update(meta_info)
 719        timing_generate = {}
 720        with self.rollout_sharding_manager:
 721            log_gpu_memory_usage("After entering rollout sharding manager", logger=logger)
 722
 723            prompts = self.rollout_sharding_manager.preprocess_data(prompts)  # ⭐ Preprocess the prompts
 724            with _timer("generate_sequences", timing_generate):
 725                output = self.rollout.generate_sequences(prompts=prompts)  # ⭐ Generate sequences based on the prompts
 726
 727            log_gpu_memory_usage("After rollout generation", logger=logger)

 729            output = self.rollout_sharding_manager.postprocess_data(output)  # ⭐ Postprocess the generated sequences

 732        # We calculate the average timing across all ranks

 734        timing_generate = reduce_timing(timing_generate)  # ⭐ Reduce timing information across all ranks
 735        output.meta_info["timing"] = timing_generate

 738        # clear kv cache
 739        get_torch_device().empty_cache()
 740        return output

 742    @register(dispatch_mode=Dispatch.DP_COMPUTE_PROTO)
 743    def compute_log_prob(self, data: DataProto):
 744        """
 745        Computes the log probability of the given data using the actor model, with support for LoRA and FSDP.
 746
 747        Args:
 748            data (DataProto): The input data containing tensors and meta information.
 749
 750        Returns:
 751            DataProto: The output data containing the computed old log probabilities and entropies.
 752        """
 753        # when is_lora is True, we use the actor without lora applied to calculate the log_prob
 754        # which is mostly used for ref log_prob calculation
 755        assert self._is_actor
 756        if self._is_offload_param:
 757            load_fsdp_model_to_gpu(self.actor_module_fsdp)  # ⭐ Load the FSDP model to GPU if offloading is enabled
 758
 759        # Support all hardwares
 760        from contextlib import nullcontext
 761
 762        is_lora = data.meta_info.pop("is_lora", False)
 763        adapter_ctx = self.actor.actor_module.disable_adapter() if is_lora else nullcontext()
 764        data = data.to(get_device_id())
 765        # we should always recompute old_log_probs when it is HybridEngine
 766        data.meta_info["micro_batch_size"] = self.config.rollout.log_prob_micro_batch_size_per_gpu
 767        data.meta_info["max_token_len"] = self.config.rollout.log_prob_max_token_len_per_gpu
 768        data.meta_info["use_dynamic_bsz"] = self.config.rollout.log_prob_use_dynamic_bsz
 769        data.meta_info["temperature"] = self.config.rollout.temperature
 770        # perform recompute log_prob
 771        with self.ulysses_sharding_manager:
 772            data = self.ulysses_sharding_manager.preprocess_data(data)
 773            with adapter_ctx:
 774                output, entropys = self.actor.compute_log_prob(data=data, calculate_entropy=True)  # ⭐ Compute log probabilities and entropies
 775            output = DataProto.from_dict(
 776                tensors={"old_log_probs": output, "entropys": entropys},
 777                meta_info={"temperature": self.config.rollout.temperature},
 778            )
 779            output = self.ulysses_sharding_manager.postprocess_data(output)
 780
 781        output = output.to("cpu")
 782
 783        # https://pytorch.org/docs/stable/notes/fsdp.html#fsdp-notes
 784        # unshard the root FSDP module
 785        if self.world_size > 1 and fsdp_version(self.actor.actor_module) == 1:
 786            self.actor.actor_module._handle.reshard(True)  # ⭐ Unshard the FSDP module if in a distributed setup
 787
 788        if self._is_offload_param:
 789            offload_fsdp_model_to_cpu(self.actor_module_fsdp)  # ⭐ Offload the FSDP model to CPU if offloading is enabled
 790            log_gpu_memory_usage("After offload actor model during compute_log_prob", logger=logger)
 791
 792        return output

429    def _build_rollout(self, trust_remote_code=False):
430        """
431        Builds the rollout and its associated sharding manager based on the configuration.
432
433        Args:
434            trust_remote_code (bool): Whether to trust remote code or not.
435

437            tuple: A tuple containing the rollout and its sharding manager.

440
441        # TODO(sgm): support FSDP hybrid shard for larger model
442        infer_tp = self.config.rollout.tensor_model_parallel_size
443        dp = self.world_size // infer_tp
444        assert self.world_size % infer_tp == 0, f"rollout world_size: {self.world_size} is not divisible by infer_tp: {infer_tp}"
445        rollout_device_mesh = init_device_mesh(device_name, mesh_shape=(dp, infer_tp), mesh_dim_names=["dp", "infer_tp"])  # ⭐ Initialize the device mesh for the rollout

451            rollout = HFRollout(module=self.actor_module_fsdp, config=self.config.rollout)
452            rollout_sharding_manager = BaseShardingManager()
453            # TODO: a sharding manager that do nothing?
454
455        elif rollout_name == "vllm":
456            from verl.workers.rollout.vllm_rollout import vllm_mode, vLLMRollout
457            from verl.workers.sharding_manager.fsdp_vllm import FSDPVLLMShardingManager
458
459            log_gpu_memory_usage(f"Before building {rollout_name} rollout", logger=logger)

466                from verl.workers.rollout.vllm_rollout import vLLMAsyncRollout

469                rollout = vllm_rollout_cls(model_path=local_path, config=self.config.rollout, tokenizer=self.tokenizer, model_hf_config=self.actor_model_config, device_mesh=rollout_device_mesh, trust_remote_code=trust_remote_code, **lora_kwargs)  # ⭐ Initialize the vLLM rollout
470            else:
471                raise NotImplementedError("vllm_mode must be 'customized' or 'spmd'")
472
473            log_gpu_memory_usage(f"After building {rollout_name} rollout", logger=logger)

490                    "'sglang_async' has been deprecated and merged into 'sglang'. Please use 'sglang' going forward.",
491                    DeprecationWarning,
492                    stacklevel=2,
493                )
494            from verl.workers.rollout.sglang_rollout import SGLangRollout
495
496            # NOTE(linjunrong): Due to recent fp8 support in SGLang. Now importing any symbol relate to
497            # SGLang's model_runner would check CUDA device capability. However, due to verl's setting,
498            # the main process of ray can not find any CUDA device, which would potentially lead to:
499            # "RuntimeError: No CUDA GPUs are available".

503            from verl.workers.sharding_manager.fsdp_sglang import FSDPSGLangShardingManager
504

506            log_gpu_memory_usage(f"Before building {rollout_name} rollout", logger=logger)
507            rollout = SGLangRollout(
508                actor_module=local_path,
509                config=self.config.rollout,
510                tokenizer=self.tokenizer,
511                model_hf_config=self.actor_model_config,
512                trust_remote_code=trust_remote_code,
513            )
514            log_gpu_memory_usage(f"After building {rollout_name} rollout", logger=logger)
515
516            if torch.distributed.get_world_size() == 1:
517                self.config.rollout.load_format = "dummy_hf"

519                module=self.actor_module_fsdp,

529            raise NotImplementedError(f"Rollout name: {self.config.rollout.name} is not supported")
530
531        return rollout, rollout_sharding_manager

646    @register(dispatch_mode=Dispatch.DP_COMPUTE_PROTO)
647    def update_actor(self, data: DataProto):
648        """
649        Updates the actor model in a distributed training setup using Fully Sharded Data Parallel (FSDP).
650
651        Args:
652            data (DataProto): The input data for the actor model.
653
654        Returns:
655            DataProto: The output data containing the updated metrics.
656        """
657        # Support all hardwares
658        data = data.to("cpu")  # data will to device with each micro batch on actor.update_policy
659
660        assert self._is_actor
661        if self._is_offload_param:
662            load_fsdp_model_to_gpu(self.actor_module_fsdp)  # ⭐ Load the FSDP model to GPU
663        if self._is_offload_optimizer:
664            load_fsdp_optimizer(optimizer=self.actor_optimizer, device_id=get_device_id())  # ⭐ Load the FSDP optimizer to GPU
665
666        with self.ulysses_sharding_manager:
667            data = self.ulysses_sharding_manager.preprocess_data(data=data)
668            # perform training
669            with Timer(name="update_policy", logger=None) as timer:
670                metrics = self.actor.update_policy(data=data)  # ⭐ Update the actor policy with the provided data
671            delta_time = timer.last
672            global_num_tokens = data.meta_info["global_token_num"]
673            estimated_flops, promised_flops = self.flops_counter.estimate_flops(global_num_tokens, delta_time)
674            metrics["perf/mfu/actor"] = estimated_flops * self.config.actor.ppo_epochs / promised_flops / self.world_size
675            metrics["perf/max_memory_allocated_gb"] = get_torch_device().max_memory_allocated() / (1024**3)
676            metrics["perf/max_memory_reserved_gb"] = get_torch_device().max_memory_reserved() / (1024**3)
677            metrics["perf/cpu_memory_used_gb"] = psutil.virtual_memory().used / (1024**3)
678
679            lr = self.actor_lr_scheduler.get_last_lr()[0]
680            metrics["actor/lr"] = lr
681            self.actor_lr_scheduler.step()
682
683            # TODO: here, we should return all metrics
684            output = DataProto(meta_info={"metrics": metrics})
685
686            output = self.ulysses_sharding_manager.postprocess_data(data=output)
687            output = output.to("cpu")
688
689        if self._is_offload_param:
690            offload_fsdp_model_to_cpu(self.actor_module_fsdp)  # ⭐ Offload the FSDP model to CPU
691            log_gpu_memory_usage("After offload actor model during update_actor", logger=logger)
692        if self._is_offload_optimizer:
693            offload_fsdp_optimizer(optimizer=self.actor_optimizer)  # ⭐ Offload the FSDP optimizer to CPU
694            log_gpu_memory_usage("After offload actor optimizer during update_actor", logger=logger)
695
696        return output

743    @register(dispatch_mode=Dispatch.DP_COMPUTE_PROTO)
744    def compute_log_prob(self, data: DataProto):
745        """
746        Computes the log probability of the given data using the actor model, with support for LoRA and FSDP.
747
748        Args:
749            data (DataProto): The input data containing tensors and meta information.
750
751        Returns:
752            DataProto: The output data containing the computed old log probabilities and entropies.
753        """
754        # when is_lora is True, we use the actor without lora applied to calculate the log_prob
755        # which is mostly used for ref log_prob calculation
756        assert self._is_actor
757        if self._is_offload_param:
758            load_fsdp_model_to_gpu(self.actor_module_fsdp)  # ⭐ Load the FSDP model to GPU if offloading is enabled
759
760        # Support all hardwares
761        from contextlib import nullcontext
762
763        is_lora = data.meta_info.pop("is_lora", False)
764        adapter_ctx = self.actor.actor_module.disable_adapter() if is_lora else nullcontext()
765        data = data.to(get_device_id())
766        # we should always recompute old_log_probs when it is HybridEngine
767        data.meta_info["micro_batch_size"] = self.config.rollout.log_prob_micro_batch_size_per_gpu
768        data.meta_info["max_token_len"] = self.config.rollout.log_prob_max_token_len_per_gpu
769        data.meta_info["use_dynamic_bsz"] = self.config.rollout.log_prob_use_dynamic_bsz
770        data.meta_info["temperature"] = self.config.rollout.temperature
771        # perform recompute log_prob
772        with self.ulysses_sharding_manager:
773            data = self.ulysses_sharding_manager.preprocess_data(data)
774            with adapter_ctx:
775                output, entropys = self.actor.compute_log_prob(data=data, calculate_entropy=True)  # ⭐ Compute log probabilities and entropies
776            output = DataProto.from_dict(
777                tensors={"old_log_probs": output, "entropys": entropys},
778                meta_info={"temperature": self.config.rollout.temperature},
779            )
780            output = self.ulysses_sharding_manager.postprocess_data(output)
781
782        output = output.to("cpu")
783
784        # https://pytorch.org/docs/stable/notes/fsdp.html#fsdp-notes
785        # unshard the root FSDP module
786        if self.world_size > 1 and fsdp_version(self.actor.actor_module) == 1:
787            self.actor.actor_module._handle.reshard(True)  # ⭐ Unshard the FSDP module if in a distributed setup
788
789        if self._is_offload_param:
790            offload_fsdp_model_to_cpu(self.actor_module_fsdp)  # ⭐ Offload the FSDP model to CPU if offloading is enabled
791            log_gpu_memory_usage("After offload actor model during compute_log_prob", logger=logger)
792
793        return output

839    @register(dispatch_mode=Dispatch.ONE_TO_ALL)

852            None
853        """
