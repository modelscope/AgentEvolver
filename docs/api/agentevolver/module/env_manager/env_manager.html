178    def step_status_printer(self, tmux):
179        """
180        Prints the current status of the steps in the parallel environment, including the number of threads in different step ranges and the token generation rate.
181
182        Args:
183            tmux (dict): A dictionary containing the 'step' and 'token' information for the current state of the environment.
184        """
185        # Initialize a counter to keep track of the number of threads in each step range
186        step_counter = {}
187
188        # Calculate the total tokens and the time elapsed since the last count
189        current_token = sum(tmux['token'])
190        current_time = time.time()
191        delta_token = current_token - self.current_token
192        delta_time = current_time - self.current_token_count_time
193        self.current_token = current_token
194        self.current_token_count_time = current_time
195        token_gen_per_sec_str = f"{delta_token/delta_time:.2f} tokens/s" if delta_time > 0 else "N/A"
196
197        # Categorize the steps into bins and count the number of threads in each bin
198        for step in tmux['step']:
199            if step == -1:
200                step_counter[(-1, 'terminated')] = step_counter.get((-1, 'terminated'), 0) + 1
201                continue
202            else:
203                start = (step // 5) * 5
204                end = start + 5
205                step_counter[(start, end)] = step_counter.get((start, end), 0) + 1
206
207        # Sort the step counter by the start value of each bin
208        step_counter = dict(sorted(step_counter.items(), key=lambda x: x[0][0]))  # ‚≠ê Sort the step counter to ensure the output is in ascending order
209
210        # Prepare the print buffer with the formatted step ranges and thread counts
211        print_buf = []
212        for (start, end), count in step_counter.items():
213            if start != -1:
214                print_buf += [f"[{start}-{end}]:{count} threads"]
215        for (start, end), count in step_counter.items():
216            if start == -1:
217                print_buf += [f"[finished]:{count} threads"]
218
219        # Print the rollout progress with the token generation rate and the step ranges
220        print(f"Rollout progress ({token_gen_per_sec_str}): " + "  //  ".join(print_buf))

281    def rollout(self, tasks: List[Task], task_exp_configs: List[TaskExpConfig], mode: Literal["sample", "validate"], epoch: str) -> List[Trajectory]:
282        """
283        Executes a list of tasks in a parallel environment using a thread pool, with automatic retries for failed tasks.
284
285        Args:
286            tasks (List[Task]): A list of tasks to be processed.
287            task_exp_configs (List[TaskExpConfig]): A list of experience configurations corresponding to each task.
288            mode (Literal["sample", "validate"]): The mode of operation, either 'sample' or 'validate'.
289            epoch (str): The current epoch identifier, used for logging and progress bar.
290
291        Returns:
292            List[Trajectory]: A sorted list of Trajectory objects representing the results of the successfully completed tasks.
293        """
294        traj_cmt_array = []
295        rollout_n = self.rollout_config.val_kwargs.n if mode == "validate" else self.rollout_n
296        future_to_params: Dict[Future, Tuple[Task, TrajExpConfig, str, str, str, int, dict, list[bool]]] = {}
297
298        tmux = {
299            'step': [0 for _ in range(len(tasks) * rollout_n)],
300            'token': [0 for _ in range(len(tasks) * rollout_n)],
301        }
302        stop = [False for _ in range(len(tasks) * rollout_n)]
303
304        with ThreadPoolExecutor(max_workers=self.max_parallel) as executor:
305            # 2. submit: submit all tasks to the thread pool
306            for data_id, (task, task_exp_config) in enumerate(zip(tasks, task_exp_configs)):
307                for rollout_id in range(rollout_n):
308                    thread_index = data_id * rollout_n + rollout_id
309                    add_exp = task_exp_config.add_exp[rollout_id]
310                    train_mode = task_exp_config.train_mode
311                    traj_exp_config = TrajExpConfig(
312                        add_exp=add_exp, train_mode=train_mode, task_id=task.task_id, data_id=data_id, rollout_id=rollout_id, mode=mode)
313
314                    params = (task, traj_exp_config, str(data_id), str(rollout_id), mode, thread_index, tmux,stop)
315                    future = executor.submit(self.rollout_env_worker, *params)
316                    future_to_params[future] = params
317
318            total_rollouts = len(future_to_params)
319            pbar = tqdm(total=total_rollouts, desc=f"Epoch {epoch}: Collecting rollouts")
320
321            # 3. wait for all tasks to complete
322            while future_to_params:
323                # if any future is done, process it
324                for future in as_completed(future_to_params):
325                    # get the corresponding params, and remove it from the dict
326                    params = future_to_params.pop(future)
327                    self.step_status_printer(tmux) # cc: i dont know what this is
328
329                    # 4. get the results and handle errors
330                    try:
331                        result = future.result()  # ‚≠ê Retrieve the result from the completed future
332
333                        # if the result has metadata error, try to recover
334                        if 'error' in result.metadata:
335                            error_msg = result.metadata['error']
336                            logger.warning(f"Task {params[1]}-{params[2]} failed with metadata error: {error_msg}. Retrying... \n Task: {params[0]}")
337                            # as most errors are internet error or quota, we wait before resubmit it
338                            time.sleep(30)
339                            # resubmit and reset tmux and stop
340                            thread_index=params[5]
341                            for k in tmux: tmux[k][thread_index] = 0
342                            stop[thread_index]=False
343                            new_future = executor.submit(self.rollout_env_worker, *params) # type: ignore
344                            future_to_params[new_future] = params
345                            continue
346
347                        # 5. if the task is successful, add it to the result list
348                        traj_cmt_array.append(result)
349                        pbar.update(1) # update progress bar when success
350
351                    except Exception as e:
352                        # handle the uncaught exception
353                        logger.error(f"Task {params[1]}-{params[2]} raised an exception: {e}. Retrying... \n Task: {params[0]}")
354                        # resubmit, and reset tmux and stop
355                        thread_index=params[5]
356                        for k in tmux: tmux[k][thread_index] = 0
357                        stop[thread_index]=False
358                        new_future = executor.submit(self.rollout_env_worker, *params) # type: ignore
359                        future_to_params[new_future] = params
360            pbar.close()
361
362        task_success_rate = np.mean([cmt.reward.success_rate for cmt in traj_cmt_array])
363        for cmt in traj_cmt_array:
364            cmt.current_batch_success_rate = np.mean(task_success_rate)
365
366        # keep trajectory sorted
367        traj_cmt_array = sorted(traj_cmt_array, key=lambda x: (int(x.data_id), int(x.rollout_id)))
368        return traj_cmt_array

409    def trajectories_to_samples(self, cmt_array: List) -> List[Sample]:
410        """
411        Converts a list of trajectories into a list of samples, ensuring the number of samples is divisible by the total number of GPUs across all nodes.
412
413        Args:
414            cmt_array (List): A list of trajectories to be converted into samples.
415
416        Returns:
417            List[Sample]: A list of samples with extras added and adjusted to be divisible by the world size.
418        """
419        # Step 1: Conversion
420        sample_arr_final = []
421        for cmt in cmt_array:
422            extras = self.get_extra(cmt)
423            # cc: message returned by the new env will be tagged as initialization, with no loss-mask
424            sample_arr = cmt.group_tokenize()  # ‚≠ê Tokenize the trajectory into samples
425            for sample in sample_arr:
426                sample.extras = extras  # ‚≠ê Add extra information to each sample
427            sample_arr_final += sample_arr
428
429        # Step 2: Calculate how many samples need to be removed
430        world_size = self.config.trainer.n_gpus_per_node * self.config.trainer.nnodes
431        remainder = len(sample_arr_final) % world_size
432        if remainder != 0:
433            import random
434            remove_indices = random.sample(range(len(sample_arr_final)), remainder)
435            # Sort in reverse order to avoid index shifting during removal
436            remove_indices.sort(reverse=True)
437            for idx in remove_indices:
438                sample_arr_final.pop(idx)  # ‚≠ê Remove samples to make the total number divisible by world size
439
440        # Randomly remove some samples, so that the number of samples is divisible by 8
441        return sample_arr_final
