127    def prepare_next_llm_context(self):
128        """
129        Prepares the next context for the LLM by filtering and modifying the latest interaction messages.
130        It removes the `think` tags from previous LLM messages, adds `/no_think` tags to non-final messages,
131        and updates the context accordingly.
132
133        Returns:
134            dict: The updated context in a dictionary format.
135        """
136        self.latest_llm_interaction_socket = []
137        # Filter out `initial message-user-llm-user-llm` or `initial message-llm-user-llm-user`
138        self.latest_llm_interaction_socket = self.filter_context_via_authors(["initialization", "llm", "env"])  #  Filter the context based on authors
139
140        for index, ext_msg in enumerate(list(self.latest_llm_interaction_socket)):
141            # is_last is the last message
142            # remove history llm author's think (and add /no_think tag to every but last message)
143            is_last = (index == len(self.latest_llm_interaction_socket) - 1)
144            # Process based on message type
145            if ext_msg.author == "llm":
146                # If it's a previous llm message, remove the think tags
147                import re
148                new_ext_msg_content = re.sub(r'<think>.*?</think>', '', ext_msg.content, flags=re.DOTALL).strip()  #  Remove <think> tags from the content
149                new_ext_msg_content = new_ext_msg_content.replace("<think>", "")
150                new_ext_msg_content = new_ext_msg_content.replace("</think>", "")
151                # new_ext_msg_content = re.sub(r'<think>.*?</think>', '<think>\n\n</think>', ext_msg.content, flags=re.DOTALL)
152                if self.config.actor_rollout_ref.rollout.train_history_infer_token:
153                    assert ext_msg.author == "llm"
154                    self.latest_llm_interaction_socket[index] = ExtendedMessage(
155                        author=ext_msg.author,
156                        role=ext_msg.role,
157                        content=new_ext_msg_content,
158                        token_generator='auto',
159                        tokenizer=self.tokenizer,
160                    )
161                else:
162                    assert ext_msg.author == "llm"
163                    author_override = "llm(do_not_train)"
164                    self.latest_llm_interaction_socket[index] = ExtendedMessage(
165                        author=author_override,
166                        role=ext_msg.role,
167                        content=new_ext_msg_content,
168                        token_generator='auto',
169                        tokenizer=self.tokenizer,
170                    )
171            elif ext_msg.author in ["env", "initialization"]:
172                if self.config.actor_rollout_ref.rollout.train_history_infer_token:
173                    # If it's initialization or environment feedback, add /no_think tag
174                    if not is_last:
175                        self.latest_llm_interaction_socket[index] = ExtendedMessage(
176                            author=ext_msg.author,
177                            role=ext_msg.role,
178                            content=ext_msg.content_for_future + "\n/no_think",
179                            token_generator='auto',
180                            tokenizer=self.tokenizer,
181                        )
182                    else:
183                        self.latest_llm_interaction_socket[index] = ExtendedMessage(
184                            author=ext_msg.author,
185                            role=ext_msg.role,
186                            content=ext_msg.content_for_future + self.think_hint,
187                            token_generator='auto',
188                            tokenizer=self.tokenizer,
189                        )
190                else:
191                    # If it's initialization or environment feedback
192                    if not is_last:
193                        self.latest_llm_interaction_socket[index] = ExtendedMessage(
194                            author=ext_msg.author,
195                            role=ext_msg.role,
196                            content=ext_msg.content_for_future,
197                            token_generator='auto',
198                            tokenizer=self.tokenizer,
199                        )
200                    else:
201                        self.latest_llm_interaction_socket[index] = ExtendedMessage(
202                            author=ext_msg.author,
203                            role=ext_msg.role,
204                            content=ext_msg.content_for_future + self.think_hint,
205                            token_generator='auto',
206                            tokenizer=self.tokenizer,
207                        )
208            else:
209                raise RuntimeError(f"Unknown author {ext_msg.author} in latest_llm_interaction_socket")
210
211        dict_context = self.to_role_content(self.latest_llm_interaction_socket)  #  Convert the filtered and modified messages to a dictionary context
212        return dict_context

216    def generate_log(self, task_id):
217        """
218        Generates a log for the given task ID, which includes tokenized steps, decoded text, and other relevant details.
219
220        Args:
221            task_id (str): The ID of the task for which the log is being generated.
222
223        Returns:
224            None
225        """
226        nested_items_print_buffer = {}
227        for index, ext_steps in enumerate(self.grouped_steps):
228            cmt_tokenized = self.tokenize_steps(ext_steps=ext_steps, debug=True)  #  Tokenize the extended steps
229            text_arr = [self.tokenizer.decode(t) for t in cmt_tokenized["input_ids"]]
230            input_id_arr = [str(t) for t in cmt_tokenized["input_ids"]]
231            loss_mask_color_arr = ["#09ABCF" if mask==1 else "#D98510" for mask in cmt_tokenized["loss_mask"]]
232            buffer = {
233                "text_arr": text_arr,
234                "input_id_arr": input_id_arr,
235                "loss_mask_color_arr": loss_mask_color_arr,
236            }
237            reward = self.reward.outcome
238            task_outcome = str(self.reward.success_rate)
239            selectors = [task_id, task_outcome, str(index)]
240            len_prompt_ids = len(cmt_tokenized["prompt_ids"])
241            len_response_ids = len(cmt_tokenized["response_ids"])
242            len_input_ids = len(cmt_tokenized["input_ids"])
243            assert len_prompt_ids + len_response_ids == len_input_ids, "len_prompt_ids + len_response_ids should equal to len_input_ids"
244            nested_items_print_buffer[f".".join(selectors)] = NestedJsonItem(
245                item_id=f"item",
246                outcome=task_outcome,
247                len_prompt_ids=len_prompt_ids,
248                len_response_ids=len_response_ids,
249                len_input_ids=len_input_ids,
250                reward=f"{float(reward):.3f}",
251                content=SeqItem(
252                    text=buffer['text_arr'],  # Text
253                    title=buffer['text_arr'],  # Hover text
254                    count=buffer['input_id_arr'],  # Highlighted text
255                    color=buffer['loss_mask_color_arr']  # Color
256                )
257            )
258        print_nested(nested_items_print_buffer,
259            main_content="This is the main content of the nested JSON",
260            header=f"Training, task {task_id}",
261            mod="rollout",
262            narrow=False,
263            attach="copy this"
264        )

272    def save_llm_output(self, llm_output, input_msg_ref):
273        """
274        Saves the LLM output and updates the internal state by appending the latest interaction to the grouped steps and resetting the latest LLM interaction socket.
275
276        Args:
277            llm_output: The output generated by the LLM.
278            input_msg_ref: A reference to the input message that triggered the LLM output.
279
280        Returns:
281            None
282        """
283        ext_msg = super().save_llm_output(llm_output, input_msg_ref)  #  Save the LLM output and get the extended message
284        this_interaction = copy.deepcopy(self.latest_llm_interaction_socket + [ext_msg])  #  Create a deep copy of the latest interaction with the new message
285        self.grouped_steps += [this_interaction]  #  Append the latest interaction to the grouped steps
286        self.latest_llm_interaction_socket = []  #  Reset the latest LLM interaction socket for the next interaction
287        return

307        """
308        Tokenizes grouped steps of a conversation and creates Sample objects for each group.
309
310        Args:
311            None
312
313        Returns:
314            list: A list of Sample objects, each representing a tokenized group of steps.
315        """
316        sample_arr = []
317        max_num_group = self.config.actor_rollout_ref.rollout.multi_turn.max_sample_per_task
318        for index, ext_steps in enumerate(self.grouped_steps):
319            if index >= max_num_group:
320                print(f"Warning: group_tokenize only process first {max_num_group} groups, but got {len(self.grouped_steps)} groups")
321                break
322            cmt_tokenized = self.tokenize_steps(ext_steps=ext_steps, debug=True)  #  Tokenize the extended steps
323            sample = Sample(
324                data_id=self.data_id,
325                rollout_id=self.rollout_id,
326                task_id=self.task_id,
327                minor_index_id=index,
328                messages=self.to_role_content(ext_steps),
329                input_ids=cmt_tokenized["input_ids"],
330                prompt_ids=cmt_tokenized["prompt_ids"],
331                response_ids=cmt_tokenized["response_ids"],
332                attention_mask=cmt_tokenized["attention_mask"],
333                prompt_attention_mask=cmt_tokenized["prompt_attention_mask"],
334                response_attention_mask=cmt_tokenized["response_attention_mask"],
335                loss_mask=cmt_tokenized["loss_mask"],
336                prompt_loss_mask=cmt_tokenized["prompt_loss_mask"],
337                response_loss_mask=cmt_tokenized["response_loss_mask"],
338                position_ids=cmt_tokenized["position_ids"],
339                prompt_position_ids=cmt_tokenized["prompt_position_ids"],
340                response_position_ids=cmt_tokenized["response_position_ids"],
341                reward_scores=self.reward.model_dump(), # reward is duplicated in each sample
342                max_prompt_len=self.config.data.max_prompt_length,
343                max_response_len=self.config.data.max_response_length,
344                max_model_len=self.config.data.max_response_length + self.config.data.max_prompt_length,
345            )
346            sample.truncate_output_ids()
347            assert len(sample.response_ids) != 0, "response_ids should not be empty"
348            sample_arr += [sample]
349        return sample_arr
